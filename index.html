2. Addition Program

#include <stdio.h>

int main() {
    int a, b;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    printf("Sum: %d\n", a + b);
    return 0;
}

3. Factorial Calculator

#include <stdio.h>

int main() {
    int n, fact = 1;
    printf("Enter number: ");
    scanf("%d", &n);
    
    for(int i = 1; i <= n; i++)
        fact *= i;
    
    printf("Factorial: %d\n", fact);
    return 0;
}
